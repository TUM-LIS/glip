37  * The two functions below are "as seen in the Linux kernel".
38  * see include/linux/math64.h and include/linux/time.h in the linux kernel tree
39  */
40 static inline uint32_t __iter_div_u64_rem(uint64_t dividend, uint32_t divisor,
41  uint64_t *remainder)
42 {
43  uint32_t ret = 0;
44 
45  while (dividend >= divisor) {
46  /* The following asm() prevents the compiler from
47  optimising this loop into a modulo operation. */
48  asm("" : "+rm"(dividend));
49  dividend -= divisor;
50  ret++;
51  }
52  *remainder = dividend;
53  return ret;
54 }

56 #define NSEC_PER_SEC 1000000000L
57 static inline void timespec_add_ns(struct timespec *a, uint64_t ns)
58 {
59  a->tv_sec += __iter_div_u64_rem(a->tv_nsec + ns, NSEC_PER_SEC, &ns);
60  a->tv_nsec = ns;
61 }
